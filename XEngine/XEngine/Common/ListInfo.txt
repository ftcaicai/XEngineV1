///类型定义
///INFOLIST(_Type,_Level,_Msg,_ShowMsg)
///_Type为信息的类型
///_Level为信息的级别
///_Msg信息简称或信息代码
///_ShowMsg信息提示

INFOLIST(ENGINERINFO,HARD,InputValueError,无效的输入值)
INFOLIST(ENGINERINFO,HARD,InputValueRangeOut,输入值超出范围)
INFOLIST(ENGINERINFO,HARD,CreateValueError,创建新的值失败)

INFOLIST(SYSTEMINFO,HARD,CreateIOCPError,创建IOCP失败)

INFOLIST(SYSTEMINFO,HARD,ODBCEnvError,初始化ODBC ENV失败)
INFOLIST(SYSTEMINFO,HARD,ODBCEnvAttrError,设置ODBC EnvAttr失败)
INFOLIST(SYSTEMINFO,HARD,ODBCDbcError,初始化ODBC DBC失败)
INFOLIST(SYSTEMINFO,HARD,ODBCStmtError,初始化ODBC STMT失败)
INFOLIST(SYSTEMINFO,NONE,ODBCInitSucceed,初始化ODBC成功)

INFOLIST(SYSTEMINFO,HARD,ODBCLinkError,连接数据库失败)
INFOLIST(SYSTEMINFO,NONE,ODBCLinkSucceed,连接数据库成功)

INFOLIST(SYSTEMINFO,HARD,CreateODBCExecHandleError,创建ODBC SQL执行句柄失败)
